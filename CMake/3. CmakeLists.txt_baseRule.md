# 基本语法
- cmake_minimum_required(VERSION 3.16)  #定义cmake的版本号
- project(demo)  #定义项目名称
- add_executable(demo main.cpp)  #定义可执行文件名称 和 参与编译的源文件；把源文件编译成可执行文件
- add_library(demo STATIC main.cpp)  #定义库文件名称 和 参与编译的源文件；把源文件编译成静态库

# 添加源文件
- aux_source_directory(dir var)  #将指定目录下的所有源文件添加到变量中,这里把dir目录下的所有源文件添加到var变量中, var 常命名为 SRC_LIST;弊端就是会把所有文件都添加到变量中，可能会加入错误文件。可以使用set命令来过滤文件。
- set(var value1 value2 ...) #定义变量var存放值value1，value2，可以定义多个值，用空格隔开

# 添加头文件搜索路径
- include_directories(dir1 dir2 ...)  #添加头文件搜索路径, 多个路径用空格隔开
- target_include_directories(target [SYSTEM] [BEFORE] dir1 dir2 ...)  #添加头文件搜索路径，target为目标名称,dir为目录
    - target_include_directories(target PRIVATE dir1)  # PRIVATE 关键字表示只在编译 target 时包含这些目录，不会影响链接到该目标的可执行文件或库。
    - target_include_directories(target INTERFACE dir1)  # INTERFACE 关键字表示只在编译使用该目标的可执行文件或库时包含这些目录，不会影响目标本身的编译。
    - target_include_directories(target PUBLIC dir1)  # PUBLIC 关键字表示在编译目标本身和链接到该目标的可执行文件或库时都包含这些目录。

# 增加编译子目录
- add_subdirectory(subdir [binary_dir] [EXCLUDE_FROM_ALL])  #增加编译子目录，subdir指定一个包含 CMakeLists.txt 文件的目录，该目录可以是绝对路径，也可以是相对路径。如果是相对路径，起点是CMAKE_CURRENT_SOURCE_DIR。binary_dir 是编译子目录的二进制文件存放目录，默认为子目录的二进制文件存放目录。EXCLUDE_FROM_ALL 参数的作用是，将子目录从编译列表中排除。

# 设置目标二进制可执行文件存放位置
- set(EXECUTABLE_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/bin) #设置目标二进制可执行文件存放在工程根目录下的bin目录下

# 系统预定义变量
- CMAKE_SOURCE_DIR  #项目的根目录，即包含顶层CMakeLists.txt的目录
- CMAKE_CURRENT_SOURCE_DIR  #当前CMakeLists.txt所在目录
- CMAKE_BINARY_DIR  #构建树中的顶层目录（通常是执行cmake命令时所在目录）
- CMAKE_CURRENT_BINARY_DIR  #当前正在处理的构建目录
- CMAKE_PROJECT_NAME  #项目名称(通过project（）命令定义)
- PROJECT_SOURCE_DIR  #工程根目录

# 查找并加载一个包
- find_package(PackageName [version] [EXACT] [QUIET] [MODULE])  #查找并加载一个包，PackageName为要查找的包的名称。
    - version 指定版本号；
    - EXACT 要求精确匹配指定的版本号；
    - QUIET 如果找不到包则不输出任何信息；
    - REQUIRED 如果找不到包，则CMake将停止处理并报错；
    - MODULE 只搜索 FindPackageName.cmake 文件（不使用 PackageNameConfig.cmake）。

# 设置编译选项
- add_compile_options(-Wall -Wextra) #添加编译选项, 多个用空格隔开
- target_compile_options(target PRIVATE -Wall -Wextra)  #为目标设置编译选项
